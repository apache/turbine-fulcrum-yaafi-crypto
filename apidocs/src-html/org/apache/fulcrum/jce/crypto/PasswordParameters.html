022/**
023 * Parameters for creating a password.
024 *
025 * @author <a href="mailto:siegfried.goeschl@it20one.at">Siegfried Goeschl</a>
026 */
027
028public interface PasswordParameters
029{
030    /** Parameter for the number of SHA256 invocation */
031    int COUNT = 1000;
032
033    /** 
034     * The default password used for creating the internal password 
035     * @return the default password: <code>fulcrum-yaafi</code>.
036     * */
037    public static char[] DefaultPassword() { 
038        return new char[] {
039        (char) 'f', (char) 'u', (char) 'l', (char) 'c',
040        (char) 'r', (char) 'u', (char) 'm', (char) '-',
041        (char) 'y', (char) 'a', (char) 'a', (char) 'f',
042        (char) 'i'
043        };
044    }
045
046    /** The password salt 
047     * @return the 8bit default salt as byte array
048     * */
049    public static byte[] Salt() {
050        return new byte[] {
051        (byte)0xc6, (byte)0x74, (byte)0x81, (byte)0x8a,
052        (byte)0x7b, (byte)0xe8, (byte)0xfe, (byte)0x99
053        };
054    }
055}
