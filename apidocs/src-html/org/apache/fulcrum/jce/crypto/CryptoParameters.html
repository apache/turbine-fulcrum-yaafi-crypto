023/**
024 * CryptoParameters used for encryption/decrytpion.
025 *
026 * @author <a href="mailto:siegfried.goeschl@it20one.at">Siegfried Goeschl</a>
027 */
028
029public interface CryptoParameters
030{
031    /** Parameter for PBEParameterSpec */
032    int COUNT = 20;
033
034    /** 
035     * The password salt: update to a method to prevent malicious code bug.
036     * 
037     * @return the salt 
038     * 
039     * */
040    public static byte[] Salt() 
041    {
042        return new byte[] 
043                        {
044                                        (byte)0xc7, (byte)0x73, (byte)0x21, (byte)0x8c,
045                                        (byte)0x7e, (byte)0xc8, (byte)0xee, (byte)0x99
046                        };
047    }
048    
049    
050    /** 
051     * The default crypto algorithm being used */
052    String ALGORITHM = "PBEWithMD5AndDES";}
