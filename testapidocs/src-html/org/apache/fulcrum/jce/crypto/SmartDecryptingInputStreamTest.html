036public class SmartDecryptingInputStreamTest extends TestCase
037{
038    /** the password to be used */
039    private String password;
040
041    /** the test data directory */
042    private File testDataDirectory;
043
044    /** the temp data director */
045    private File tempDataDirectory;
046
047    /**
048     * Constructor
049     * @param name the name of the test case
050     */
051    public SmartDecryptingInputStreamTest( String name )
052    {
053        super(name);
054
055        this.password = "mysecret";
056        this.testDataDirectory = new File( "./src/test/data" );
057        this.tempDataDirectory = new File( "./temp" );
058    }
059
060    /**
061     * @see junit.framework.TestCase#setUp()
062     */
063    protected void setUp() throws Exception
064    {
065        CryptoStreamFactoryImpl factory = new CryptoStreamFactoryImpl(
066            CryptoParameters.Salt(),
067            CryptoParameters.COUNT
068            );
069
070        CryptoStreamFactoryImpl.setInstance( factory );
071    }
072
073    /**
074     * @return Returns the password.
075     */
076    protected char[] getPassword()
077    {
078        return password.toCharArray();
079    }
080
081    /**
082     * @return Returns the tempDataDirectory.
083     */
084    protected File getTempDataDirectory()
085    {
086        return tempDataDirectory;
087    }
088
089    /**
090     * @return Returns the testDataDirectory.
091     */
092    protected File getTestDataDirectory()
093    {
094        return testDataDirectory;
095    }

137    /** Test smart decryption for a given file */
138    private void testSmartDecryption( String fileName, String enc ) throws Exception
139    {
140        File sourceFile = new File( this.getTestDataDirectory(), fileName );
141        String plainText = this.loadTextFile(sourceFile,enc);
142        String smartText = this.smartDecrypt(sourceFile,enc);
143        byte[] cipherText = this.encryptTextFile(sourceFile);
144        String decryptedText = this.smartDecrypt(cipherText,enc);
145
146        assertTrue( plainText.length() == smartText.length() );
147        assertTrue( plainText.length() == decryptedText.length() );
148        assertEquals( plainText, smartText );
149        assertEquals( plainText, decryptedText );
150    }
151
152    /**
153     * Loads a plain text file.
154     * @param file the file to load
155     */
156    private String loadTextFile( File file, String enc ) throws Exception
157    {
158        String result = null;
159        FileInputStream fis = new FileInputStream( file );
160        ByteArrayOutputStream baos = new ByteArrayOutputStream();
161        StreamUtil.copy(fis,baos);
162        fis.close();
163        result = new String( baos.toByteArray(), enc );
164        return result;
165    }
166
167    /**
168     * Encrypt a plain text file.
169     * @param file the file to encrypt
170     */
171    private byte[] encryptTextFile( File file ) throws Exception
172    {
173        ByteArrayOutputStream baos = new ByteArrayOutputStream();
174        FileInputStream fis = new FileInputStream( file );
175
176        CryptoUtil.getInstance().encrypt(
177                CryptoStreamFactoryImpl.getInstance(),
178            fis,
179            baos,
180            this.getPassword()
181            );
182
183        fis.close();
184
185        return baos.toByteArray();
186    }
187
188    /**
189     * Use smart decryption on a cipherText.
190     *
191     * @param cipherText the encrypted text
192     * @return the decrypeted content
193     */
194    private String smartDecrypt( byte[] cipherText, String enc ) throws Exception
195    {
196        ByteArrayInputStream bais = new ByteArrayInputStream(cipherText);
197        ByteArrayOutputStream baos = new ByteArrayOutputStream();
198
199        SmartDecryptingInputStream sdis = new SmartDecryptingInputStream(
200                CryptoStreamFactoryImpl.getInstance(),
201            bais,
202            this.getPassword()
203            );
204
205        StreamUtil.copy(sdis,baos);
206
207        return new String( baos.toByteArray(), enc );
208    }
209
210    /**
211     * Use smart decryption on a plain text file.
212     *
213     * @param file the file to load
214     * @return the content
215     */
216    private String smartDecrypt( File file, String enc ) throws Exception
217    {
218        ByteArrayOutputStream baos = new ByteArrayOutputStream();
219        FileInputStream fis = new FileInputStream( file );
220
221        SmartDecryptingInputStream sdis = new SmartDecryptingInputStream(
222                CryptoStreamFactoryImpl.getInstance(),
223            fis,
224            this.getPassword()
225            );
226
227        StreamUtil.copy(sdis,baos);
228        return new String( baos.toByteArray(), enc );
229    }
