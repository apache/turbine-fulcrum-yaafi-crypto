225        @Test
226        public void testIntegratedStringEncryption() {
227                try {
228                        // assumptions
229                        String topSecret = "myX!_secretp@ssword?~,2";
230
231                        Map<String, Object> conf = new HashMap<>();
232                        conf.put("enc", TYPES.GCM);
233
234                        // encode as string to stderr and file in hexdecimal format
235                        String[] encryptionArgs = { "string", "enc" + conf.get("enc"), this.password, topSecret,
236                                        "./target/main8/integrated-plain16chars.enc.txt" };
237                        CLI2.main(encryptionArgs);
238                        // shows encoded hexdec in stdout + file
239
240                        String encodedEncrypted = FileUtils.readFileToString(
241                                        new File("./target/main8/integrated-plain16chars.enc.txt"), StandardCharsets.UTF_8);
242
243                        conf.put("pw", encodedEncrypted);
244
245                        // this should be done without output to console
246                        String result = CLI2.processString("dec" + conf.get("enc"), this.password.toCharArray(),
247                                        (String) conf.get("pw"));
248
249                        assertTrue(result.equals(topSecret));
250
251                } catch (Exception e) {
252                        e.printStackTrace();
253                        fail();
254                }
255        }

257        @Test
258    public void testDefaultStringEncryption() {
259        char[] testVector = new char[513];
260
261        for (int i = 0; i < testVector.length; i++) {
262            testVector[i] = (char) i;
263        }
264        String source = new String(testVector);
265       
266        CryptoUtilJ8.getInstances().clear();
267        CryptoUtilJ8 cuj8 = CryptoUtilJ8.getInstance();
268        String cipherText;
269        String plainText;
270        try {
271            cipherText = cuj8.encryptString(source, this.getPassword());
272            plainText = cuj8.decryptString(cipherText, this.getPassword());
273            assertEquals(source, plainText, source +" is not equal with " + plainText); 
274        } catch (GeneralSecurityException | IOException e) {
275            e.printStackTrace();
276            fail();
277        }
278    }
