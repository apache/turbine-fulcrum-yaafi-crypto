081    @BeforeAll
082    public static void setUp() throws Exception {
083        cryptoUtilJ8s.clear();
084        SupportedTypeArguments.init();
085        for (TYPES type : CryptoParametersJ8.TYPES.values()) {
086            if (SupportedTypeArguments.SUPPORTED_TYPES.contains(type.toString())) {
087                cryptoUtilJ8s.add(CryptoUtilJ8.getInstance(type));
088            }
089        }
090        for (CryptoUtilJ8 cryptoUtilJ8 : cryptoUtilJ8s) {
091            log.debug("registered {}: {}", cryptoUtilJ8.getClass().getSimpleName());
092            CryptoStreamFactoryJ8Template crt = ((CryptoStreamFactoryJ8Template)cryptoUtilJ8.getCryptoStreamFactory());
093            log.debug(String.format("created default crypto factory instance %s for algo %s with salt length: %s", 
094                        crt.getClass().getSimpleName(),
095                        crt.getAlgorithm(), crt.getSalt().length));
096        }
097    }

148    /** Decrypt a text file 
149     */
150    @Test
151    public void testTextDecryption() {           
152        cryptoUtilJ8s.forEach(cuj8 -> { 
153                log.debug("registered {}: {}", cuj8.getClass().getSimpleName());
154            CryptoStreamFactoryJ8Template crt = ((CryptoStreamFactoryJ8Template)cuj8.getCryptoStreamFactory());
155            log.debug(String.format("created default crypto factory instance %s for algo %s with salt length: %s", 
156                        crt.getClass().getSimpleName(),
157                        crt.getAlgorithm(), crt.getSalt().length));
158                try {
159                    File sourceFile = new File(this.getTestDataDirectory(), "plain.txt");
160                    File targetFile = new File(this.getTempDataDirectory(), "plain.j8.enc.txt");
161                    cuj8.encrypt(sourceFile, targetFile, this.getPassword());
162                    
163                    File sourceFile2 = new File(this.getTempDataDirectory(), "plain.j8.enc.txt");;
164                    File targetFile2 = new File(this.getTempDataDirectory(), "plain.j8.dec.txt");
165                    cuj8.decrypt(sourceFile2, targetFile2.getAbsolutePath(), this.getPassword());
166                    assertEquals(
167                            new String(Files.readAllBytes( Paths.get(sourceFile.toURI())) ), 
168                            new String(Files.readAllBytes( Paths.get(targetFile2.toURI())) )
169                            );
170                } catch (GeneralSecurityException | IOException e) {
171                    e.printStackTrace();
172                    fail();
173                }
174            });
175    }

286    /** Test encryption and decryption of binary data
287     * @throws Exception Generic exception
288     */
289    @Test
290    public void testBinaryHandling() throws Exception {
291        
292        cryptoUtilJ8s.forEach(cuj8 -> { 
293            byte[] source = new byte[256];
294            byte[] result = null;
295
296            for (int i = 0; i < source.length; i++) {
297                source[i] = (byte) i;
298            }
299
300            ByteArrayOutputStream cipherText = new ByteArrayOutputStream();
301            ByteArrayOutputStream plainText = new ByteArrayOutputStream();
302            try {
303                cuj8.encrypt(source, cipherText, this.getPassword());
304                cuj8.decrypt(cipherText, plainText, this.getPassword());
305            } catch (GeneralSecurityException | IOException e) {
306                e.printStackTrace();
307                fail();
308            }
309            result = plainText.toByteArray();
310
311            for (int i = 0; i < source.length; i++) {
312                if (source[i] != result[i]) {
313                    fail("Binary data are different at position " + i);
314                }
315            }
316        });

322    /** 
323     * Test encryption and decryption of Strings 
324     */
325    @Test
326    public void testStringWithPasswordEncryption() {
327        char[] password = "57cb-4a23-d838-45222".toCharArray();
328        String source = "e02c-3b76-ff1e-5d9a1";
329        
330        cryptoUtilJ8s.forEach(cuj8 -> { 
331            String cipherText = null;
332            try {
333                cipherText = cuj8.encryptString(source, password);
334                log.debug(cipherText);// about 128
335                
336                CryptoStreamFactoryJ8Template crt = ((CryptoStreamFactoryJ8Template)cuj8.getCryptoStreamFactory());
337                log.debug(String.format("created default crypto factory instance %s for algo %s and type %s with salt (optional): %s", 
338                                crt.getClass().getSimpleName(), crt.getType(),
339                                crt.getAlgorithm(), crt.getSalt()));
340
341                log.debug("length for {} is: {}", crt.getType(), cipherText.length());// about 128
342                if (crt.getType() == TYPES.PBE) {
343                    assertEquals(128, cipherText.length()); // 128bytes + 10 bytes for cleartext
344                } 
345                CryptoStreamFactoryJ8Template.resetInstances();
346                String plainText = cuj8.decryptString(cipherText, password);
347                assertEquals(source, plainText);
348            } catch (GeneralSecurityException | IOException e) {
349                e.printStackTrace();
350                fail();
351            }
352            
353        });
