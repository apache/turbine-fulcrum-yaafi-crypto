073    protected static byte[] generateSalt() {
074        SecureRandom random;
075        try {
076            random = SecureRandom.getInstanceStrong();
077            byte[] salt = new byte[ 16 ];
078            random.nextBytes(salt);
079            return salt;
080        } catch (NoSuchAlgorithmException e) {
081            e.printStackTrace();
082        }
083        return null;
084    }

098    @BeforeAll
099    public static void setUp() throws Exception {
100        cryptoUtilJ8s.clear();
101        SupportedTypeArguments.init();
102        for (TYPES type : CryptoParametersJ8.TYPES.values()) {
103            if (SupportedTypeArguments.SUPPORTED_TYPES.contains(type.toString())) {
104                cryptoUtilJ8s.add(CryptoUtilJ8.getInstance(type, SALT, COUNT));
105            }
106        }
107        for (CryptoUtilJ8 cryptoUtilJ8 : cryptoUtilJ8s) {
108            log.debug("registered {}", cryptoUtilJ8.getClass().getSimpleName() );
109            CryptoStreamFactoryJ8Template crt = ((CryptoStreamFactoryJ8Template)cryptoUtilJ8.getCryptoStreamFactory());
110            log.debug(String.format("created default crypto factory instance %s for algo %s with salt length: %s", 
111                        crt.getClass().getSimpleName(),
112                        crt.getAlgorithm(), crt.getSalt().length));
113        }
114    }

165    /** Decrypt a text file 
166     */
167    @Test
168    public void testTextDecryption() {           
169        cryptoUtilJ8s.forEach(cuj8 -> { 
170            log.info("start en-/decrypting with {}",cuj8);    
171                try {
172                    File sourceFile = new File(this.getTestDataDirectory(), "plain.txt");
173                    File targetFile = new File(this.getTempDataDirectory(), "plain.j8.enc.txt");
174                    cuj8.encrypt(sourceFile, targetFile, this.getPassword());

176                    File sourceFile2 = new File(this.getTempDataDirectory(), "plain.j8.enc.txt");;
177                    File targetFile2 = new File(this.getTempDataDirectory(), "plain.j8.dec.txt");
178                    cuj8.decrypt(sourceFile2, targetFile2.getAbsolutePath(), this.getPassword());
179                    assertEquals(
180                            new String(Files.readAllBytes( Paths.get(sourceFile.toURI())) ), 
181                            new String(Files.readAllBytes( Paths.get(targetFile2.toURI())) )
182                            );
183                } catch (GeneralSecurityException | IOException e) {
184                    e.printStackTrace();
185                    fail();
186                }
187            });

299    /** Test encryption and decryption of binary data
300     * @throws Exception Generic exception
301     */
302    @Test
303    public void testBinaryHandling() throws Exception {
304        
305        cryptoUtilJ8s.forEach(cuj8 -> { 
306            byte[] source = new byte[256];
307            byte[] result = null;
308
309            for (int i = 0; i < source.length; i++) {
310                source[i] = (byte) i;
311            }
312
313            ByteArrayOutputStream cipherText = new ByteArrayOutputStream();
314            ByteArrayOutputStream plainText = new ByteArrayOutputStream();
315            try {
316                cuj8.encrypt(source, cipherText, this.getPassword());
317                cuj8.decrypt(cipherText, plainText, this.getPassword());
318            } catch (GeneralSecurityException | IOException e) {
319                e.printStackTrace();
320                fail();
321            }
322            result = plainText.toByteArray();
323
324            for (int i = 0; i < source.length; i++) {
325                if (source[i] != result[i]) {
326                    fail("Binary data are different at position " + i);
327                }
328            }
329        });

335    /** 
336     * Test encryption and decryption of Strings 
337     */
338    @Test
339    public void testStringWithPasswordEncryption() {
340        char[] password = "57cb-4a23-d838-45222".toCharArray();
341        String source = "e02c-3b76-ff1e-5d9a1";
342        
343        cryptoUtilJ8s.forEach(cuj8 -> { 
344            String cipherText = null;
345            try {
346                cipherText = cuj8.encryptString(source, password);
347                log.debug(cipherText);// about 128
348                
349                log.debug("registered {}: {}", cuj8.getClass().getSimpleName());
350                CryptoStreamFactoryJ8Template crt = ((CryptoStreamFactoryJ8Template)cuj8.getCryptoStreamFactory());
351                log.debug(String.format("created default crypto factory instance %s for algo %s with salt (optional): %s", 
352                        crt.getClass().getSimpleName(),
353                        crt.getAlgorithm(), crt.getSalt()));
354                
355                log.debug("length for {} is: {}", crt.getType(), cipherText.length());// about 128
356                if (crt.getType() == TYPES.PBE) {
357                    assertEquals(128, cipherText.length()); // 128bytes + 10 bytes for cleartext
358                } 
359                CryptoStreamFactoryJ8Template.resetInstances();
360                String plainText = cuj8.decryptString(cipherText, password);
361                assertEquals(source, plainText);
362            } catch (GeneralSecurityException | IOException e) {
363                e.printStackTrace();
364                fail();
365            }
366            
367        });
