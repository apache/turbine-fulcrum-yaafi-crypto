106

118                                cuj8.encrypt(sourceFile, targetFile, this.getPassword());
119                        } catch (GeneralSecurityException e) {
120                                e.printStackTrace();
121                                fail();
122                        } catch (IOException e) {
123                                e.printStackTrace();
124                                fail();
125                        }
126                });

217        /**
218         * Parameterized Test Test encryption and decryption of Strings
219         * 
220         * @param type the type to be tested based on {@link TYPES}
221         * 
222         */
223        @ParameterizedTest
224        @ArgumentsSource(SupportedTypeArguments.class)
225        public void testStringEncryption(TYPES type) {
226                CryptoUtilJ8 cuj8 = CryptoUtilJ8.getInstance(type);
227                
228                log.debug("registered {} and called for {}", cuj8.getClass().getSimpleName(), type);
229        CryptoStreamFactoryJ8Template crt = ((CryptoStreamFactoryJ8Template)cuj8.getCryptoStreamFactory());
230        log.debug(String.format("created default crypto factory instance %s for algo %s with salt length: %s", 
231                        crt.getClass().getSimpleName(),
232                        crt.getAlgorithm(), crt.getSalt().length));
233                char[] testVector = new char[513];
234
235                for (int i = 0; i < testVector.length; i++) {
236                        testVector[i] = (char) i;
237                }
238
239                String source = new String(testVector);
240                String cipherText = null;
241                String plainText = null;
242                try {
243                        log.info("Test without clearTextHeader in type {}", type);
244                        cipherText = cuj8.encryptString(source, this.getPassword());
245                        log.trace(cipherText);
246                        plainText = cuj8.decryptString(cipherText, this.getPassword());
247                        assertEquals(source, plainText, source + " is not equal with " + plainText);
248                        
249                        String clearCode = type.equals(TYPES.PBE)? CryptoParametersJ8.TYPES.PBE.getClearCode():
250                                CryptoParametersJ8.TYPES.GCM.getClearCode() ;
251
252                        log.info(String.format("Test with clearTextHeader %s in encrypted string.",clearCode) );
253                        String cipherText2 = cuj8.encryptStringWithClearCode(source, this.getPassword());
254                        log.trace(cipherText2);
255                        assertTrue(cipherText2.startsWith(clearCode),
256                                        String.format("%s does not start with '%s'", cipherText2, clearCode));
257                        String plainText2 = cuj8.decryptStringWithClearCode(cipherText2, this.getPassword());
258                        assertEquals(source, plainText2, String.format("%s is not equal with %s", source, plainText));
259
260                } catch (GeneralSecurityException | IOException e) {
261                        e.printStackTrace();
262                        fail();
263                }
264        }

318        @ArgumentsSource(SupportedTypeArguments.class)
319        public void testBinaryHandling(TYPES type) throws Exception {
320                cryptoUtilJ8s.add(CryptoUtilJ8.getInstance(type));
321                cryptoUtilJ8s.forEach(cuj8 -> {
322                        byte[] source = new byte[256];
323                        byte[] result = null;
324
325                        for (int i = 0; i < source.length; i++) {
326                                source[i] = (byte) i;
327                        }
328
329                        ByteArrayOutputStream cipherText = new ByteArrayOutputStream();
330                        ByteArrayOutputStream plainText = new ByteArrayOutputStream();
331                        try {
332                                cuj8.encrypt(source, cipherText, this.getPassword());
333                                cuj8.decrypt(cipherText, plainText, this.getPassword());
334                        } catch (GeneralSecurityException | IOException e) {
335                                e.printStackTrace();
336                                fail();
337                        }
338                        result = plainText.toByteArray();
339
340                        for (int i = 0; i < source.length; i++) {
341                                if (source[i] != result[i]) {
342                                        fail("Binary data are different at position " + i);
343                                }
344                        }
345                });
