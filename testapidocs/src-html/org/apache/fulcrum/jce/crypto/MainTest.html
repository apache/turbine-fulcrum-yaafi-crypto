024import junit.framework.TestCase;
025
026/**
027 * Test suite for crypto functionality
028 *
029 * @author <a href="mailto:siegfried.goeschl@it20one.at">Siegfried Goeschl</a>
030 */
031
032public class MainTest extends TestCase
033{
034    /** the password to be used */
035    private String password;
036
037    /**
038     * Constructor
039     * @param name the name of the test case
040     */
041    public MainTest( String name )
042    {
043        super(name);
044
045        this.password = "foobar";
046    }
047
048    /**
049     * @return Returns the password.
050     */
051    protected char[] getPassword()
052    {
053        return password.toCharArray();
054    }
055
056    /** Encrypt a string on the command line */
057    public void testStringEncryption()
058    {
059        String[] encryptionArgs = { "string", "enc", this.password, "mysecretpassword"};
060        CLI.main(encryptionArgs);
061        String[] decryptionArgs = { "string", "dec", this.password, "9330419fc003b4e1461986782625db13f4c8c81c340a9caa"};
062        CLI.main(decryptionArgs);
063    }
064    
065    public void testAnotherStringEncryption()
066    {
067        String[] encryptionArgs = { "string", "enc", this.password, "secret"};
068        CLI.main(encryptionArgs);
069        String[] decryptionArgs = { "string", "dec", this.password, "39619852d48491af"};
070        CLI.main(decryptionArgs);
071    }
072
073    /** Encrypt a text file on the command line */
074    public void testFileEncryption1()
075    {
076        String[] encryptionArgs = { "file", "enc", this.password, "./src/test/data/plain.txt", "./target/main/plain.enc.txt" };
077        String[] decryptionArgs = { "file", "dec", this.password, "./target/main/plain.enc.txt", "./target/main/plain.dec.txt" };
078        CLI.main(encryptionArgs);
079        CLI.main(decryptionArgs);
080    }
081
082    /** Encrypt a text file in-place on the command line */
083    public void testFileEncryption2()
084    {
085        String[] encryptionArgs = { "file", "enc", this.password, "./src/test/data/plain.txt", "./target/main/plain.txt" };
086        String[] decryptionArgs = { "file", "dec", this.password, "./target/main/plain.txt" };
087        CLI.main(encryptionArgs);
088        CLI.main(decryptionArgs);
089    }
090
091}
