317        @Test
318        public void testStringEncryptionWithType() {
319                CryptoUtil cu = CryptoUtil.getInstance();
320                char[] testVector = new char[513];
321
322                for (int i = 0; i < testVector.length; i++) {
323                        testVector[i] = (char) i;
324                }
325
326                String source = new String(testVector);
327                String cipherText = null;
328                String plainText = null;
329                try {
330                        log.info("Test without clearTextHeader");
331                        cipherText = cu.encryptString(source, this.getPassword());
332                        log.trace(cipherText);
333                        plainText = cu.decryptString(cipherText, this.getPassword());
334                        assertEquals(source, plainText, source + " is not equal with " + plainText);
335
336                        log.info(String.format("Test with clearTextHeader %s in encrypted string.",
337                                        CryptoParametersJ8.CLEAR_CODE_DEFAULT));
338                        String cipherText2 = cu.encryptStringWithClearCode(source, this.getPassword());
339                        log.trace(cipherText2);
340                        // old style
341                        assertTrue(cipherText2.startsWith(CryptoParametersJ8.CLEAR_CODE_DEFAULT),
342                                        String.format("%s does not start with '%s'", cipherText2, CryptoParametersJ8.CLEAR_CODE_DEFAULT));
343                        String plainText2 = cu.decryptStringWithClearCode(cipherText2, this.getPassword());
344                        assertEquals(source, plainText2, String.format("%s is not equal with %s", source, plainText));
345
346                } catch (GeneralSecurityException | IOException e) {
347                        e.printStackTrace();
348                        fail();
349                }
350        }

367        /** Test encryption and decryption of binary data
368         * @throws Exception Generic exception
369         */
370         @Test
371        public void testBinaryHandling() throws Exception {
372                byte[] source = new byte[256];
373                byte[] result = null;
374
375                for (int i = 0; i < source.length; i++) {
376                        source[i] = (byte) i;
377                }
378
379                ByteArrayOutputStream cipherText = new ByteArrayOutputStream();
380                ByteArrayOutputStream plainText = new ByteArrayOutputStream();
381
382                CryptoUtil.getInstance().encrypt(source, cipherText, this.getPassword());
383                CryptoUtil.getInstance().decrypt(cipherText, plainText, this.getPassword());
384
385                result = plainText.toByteArray();
386
387                for (int i = 0; i < source.length; i++) {
388                        if (source[i] != result[i]) {
389                                fail("Binary data are different at position " + i);
390                        }
391                }
392        }
