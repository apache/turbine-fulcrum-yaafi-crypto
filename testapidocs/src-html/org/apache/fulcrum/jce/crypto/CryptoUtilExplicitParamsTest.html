062         public static byte[] Salt()
063         {
064                SecureRandom random;
065                try {
066                        random = SecureRandom.getInstanceStrong();
067                byte[] salt = new byte[ 8 ];
068                random.nextBytes(salt);
069                return salt;
070                } catch (NoSuchAlgorithmException e) {
071                        e.printStackTrace();
072                }
073                return null;
074         }

332        @Test
333        public void testStringEncryptionWithType() {
334                CryptoUtil cu = CryptoUtil.getInstance(SALT,COUNT);
335                char[] testVector = new char[513];
336
337                for (int i = 0; i < testVector.length; i++) {
338                        testVector[i] = (char) i;
339                }
340
341                String source = new String(testVector);
342                String cipherText = null;
343                String plainText = null;
344                try {
345                        log.info("Test without clearTextHeader");
346                        cipherText = cu.encryptString(source, this.getPassword());
347                        log.trace(cipherText);
348                        plainText = cu.decryptString(cipherText, this.getPassword());
349                        assertEquals(source, plainText, source + " is not equal with " + plainText);
350
351                        log.info(String.format("Test with clearTextHeader %s in encrypted string.",
352                                        CryptoParametersJ8.CLEAR_CODE_DEFAULT));
353                        String cipherText2 = cu.encryptStringWithClearCode(source, this.getPassword());
354                        log.trace(cipherText2);
355                        // old style
356                        assertTrue(cipherText2.startsWith(CryptoParametersJ8.CLEAR_CODE_DEFAULT),
357                                        String.format("%s does not start with '%s'", cipherText2, CryptoParametersJ8.CLEAR_CODE_DEFAULT));
358                        String plainText2 = cu.decryptStringWithClearCode(cipherText2, this.getPassword());
359                        assertEquals(source, plainText2, String.format("%s is not equal with %s", source, plainText));
360
361                } catch (GeneralSecurityException | IOException e) {
362                        e.printStackTrace();
363                        fail();
364                }
365        }

382        /** Test encryption and decryption of binary data
383         * @throws Exception Generic exception
384         */
385         @Test
386        public void testBinaryHandling() throws Exception {
387                byte[] source = new byte[256];
388                byte[] result = null;
389
390                for (int i = 0; i < source.length; i++) {
391                        source[i] = (byte) i;
392                }
393
394                ByteArrayOutputStream cipherText = new ByteArrayOutputStream();
395                ByteArrayOutputStream plainText = new ByteArrayOutputStream();
396
397                CryptoUtil.getInstance(SALT,COUNT).encrypt(source, cipherText, this.getPassword());
398                CryptoUtil.getInstance(SALT,COUNT).decrypt(cipherText, plainText, this.getPassword());
399
400                result = plainText.toByteArray();
401
402                for (int i = 0; i < source.length; i++) {
403                        if (source[i] != result[i]) {
404                                fail("Binary data are different at position " + i);
405                        }
406                }
407        }
